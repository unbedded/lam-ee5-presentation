% Table formatting for medical documents with comprehensive tables
% Designed for stress test tables, lab results, and medical data

\usepackage{array}
\usepackage{hhline}

% Table spacing and lines for readability
\renewcommand{\arraystretch}{1.3}
\setlength{\arrayrulewidth}{0.5pt}
\setlength{\tabcolsep}{6pt}

% Allow tables to break across pages
\usepackage{tabu}

% Better column formatting for medical data
\usepackage{booktabs}

% Adjust page breaking to keep tables together when possible
\usepackage{needspace}
